vocabulary V {
    type Limit isa int
    type PersonType isa string
    personType( PersonType )
    type StoreType isa string
    storeType( StoreType )
    type GameType isa string
    gameType( GameType )

    canAccess( PersonType, StoreType )

    // limits for game types per storetype
    gambleLimitsPerStore( StoreType, GameType, Limit )
    // game limit restrictions imposed up that user
    mandatedGambleLimits( GameType, Limit )
    // amount available
    account : Limit
    // to add limit 0
    fullLimited( Limit )
    // what limits apply for this situation
    applicableLimits( PersonType, GameType, Limit )

    // game limit restrictions set by user
    personalGambleLimits( GameType, Limit )

    // output limits
    gambleLimits( PersonType, GameType, Limit )

    accessRequestedFor : PersonType
    accessRequestedTo : StoreType
}
