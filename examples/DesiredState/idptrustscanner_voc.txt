vocabulary V {
type InstallDateType isa int
type InstallPCType isa int
type VersionType isa int

type InstallTupleF constructed from { ITF(InstallPCType, InstallDateType, VersionType) }
type InstallType  isa InstallTupleF

lastDay : InstallDateType
lastVersion : VersionType

// state of a PC at a certain date
installstate( InstallType )
installedVersion( InstallType ) : VersionType
installDate( InstallType ) : InstallDateType
installedOn( InstallType ) : InstallPCType

// when does the the PC not comply to the required SLA
breach( InstallPCType, InstallDateType, VersionType )

// when is a certain version found first on a certain PC
partial firstDayOnPC( InstallPCType, VersionType ) : InstallDateType

// when is a certain version found last on a certain PC
partial lastDayOnPC( InstallPCType, VersionType ) : InstallDateType

// which days was that PCs at risk because breached
riskDay( InstallPCType, InstallDateType )

// which breaches were too long
longRunBreach( InstallType )

// when is deployment for a certain version started
versionInstallStartDate( VersionType ) : InstallDateType

// when is deployment for a certain version ended
partial versionInstallEndDate( VersionType ) : InstallDateType

// was a version install on a pc
wasInstalledOnPC( InstallPCType, VersionType )

// next version
partial nextVersion( VersionType ) : VersionType

// next version for a PC
//partial pcNextVersion( InstallPCType, VersionType ) : VersionType

}
