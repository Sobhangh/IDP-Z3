vocabulary V {
type InstallPCType isa int
 type NiceToHave isa int
 type Order isa int

 type PCProperty isa string
 type PCPropertyValue isa string

 type PropertyTupleF constructed from {PV(PCProperty,PCPropertyValue)}
 type PropertyTuple isa PropertyTupleF
 order(PropertyTuple) : Order
 property(PropertyTuple) : PCProperty
 value(PropertyTuple) : PCPropertyValue

 // for properties and their values
 pcProperty( InstallPCType, PropertyTuple )
 propertyForAssignedPC( InstallPCType, PropertyTuple )

 // properties that were matched to assign this PC
 matchedPropertyForAssignedPC( InstallPCType, PropertyTuple )

 // some properties require states of other properties
 propertyDependency( PCProperty, PropertyTuple )

 // some properties require states of other properties when having
 // a certain value
 propertyValueDependency( PropertyTuple, PropertyTuple )

 // lists all properties where this property depends on
 dependentProperty( PropertyTuple, PropertyTuple )

 // dependencies for this user (RBAC)
 userDependency( PropertyTuple, NiceToHave )
 userDependencyNTHThatApply( InstallPCType, PropertyTuple, NiceToHave )

 // system dependency
 systemDependency( PropertyTuple )

 // does this property hold for this PC exactly
 propertyIsStrictlyFullfilled( InstallPCType, PropertyTuple )
 // does this property hold for this PC because a higher value for this property
 // convers it
 propertyIsOverlyFullfilled( InstallPCType, PropertyTuple )
 // does this property hold for this PC
 propertyIsFullfilled( InstallPCType, PropertyTuple )

 dependentsAreFullfilled( InstallPCType, PCProperty )
 valueDependentsAreFullfilled( InstallPCType, PropertyTuple )

 // can issue this PC to current user
 canAssignPC( InstallPCType )
 cannotAssignPC( InstallPCType, PropertyTuple )
 canSimpleAssignPC( InstallPCType )

 pcNTHValue( InstallPCType ) : NiceToHave

 isNecessaryProperty( PropertyTuple )
 pcOverlyMatchingProperty( PropertyTuple )
 pcStrictlyMatchingProperty( PropertyTuple )
 pcNotMatchingProperty( PropertyTuple )


 nthValue : NiceToHave

 checkPC : InstallPCType
}
